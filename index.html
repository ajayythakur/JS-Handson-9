Qn 2. "Problem: You are given a string (STR) of length N, you have to print the count of all alphabet.(using maps)

            Output:
            a=2
            b=2
            c=2

Ans - 

                let a="abcadeecfb";
                let uniquechar=new Map();
                for(let i=0;i< a.length;i++){
                  if(uniquechar.has(a[i])){
                    let value=uniquechar.get(a[i]);
                    uniquechar.set(a[i],value+1);
                    
                  }
                  else{
                    uniquechar.set(a[i],1);
                    }
                }
                for(let [k,v] of uniquechar){
                  console.log(k+"="+ v);
                }
